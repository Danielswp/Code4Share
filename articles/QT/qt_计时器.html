2.cpp实现 
// 计时器-刷新左下角时间
TimerHandle_RefreshTime = new QTimer(this);
TimerHandle_RefreshTime->start();
TimerHandle_RefreshTime->setInterval(1000);
connect(TimerHandle_RefreshTime, &QTimer::timeout, this, &MyWindow::RefreshTime);
  
/// <summary>
/// 刷新时间文字
/// </summary>
void MyWindow::RefreshTime()
{
  QDateTime dateTime = QDateTime::currentDateTime();
  QString s_dateTime = dateTime.toString(QString::fromLocal8Bit("yyyy年MM月dd日 hh:mm:ss"));
  ui.label_myTime->setText(s_dateTime);
}
